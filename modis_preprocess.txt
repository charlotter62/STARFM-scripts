#!/bin/bash

Filename=$1
Folder=$2
Example=$3
OutLoc="/Users/crivard/Desktop/StarFM/BGR/modis_scenes/2002_2003"

Input=$Folder"/"$Filename
Output=$OutLoc"/deletes/1"$Filename
gdalwarp $Input $Output -tr 0.000269494585235856472 0.000269494585235856472

Input=$Output
Output=$OutLoc"/deletes/2"$Filename
rio clip $Input $Output --like $Example

Input=$Output
Output=$OutLoc"/deletes/3"$Filename

gdal_translate $Input $Output -b 3 -b 4 -b 1 -b 2 -b 5 -ot Int16

Input=$Output
bands="1 3 4"
#for i in {1..5}
for i in $bands
do
    Output=$OutLoc"/"$Filename"_B"$i".bin"
    gdal_translate -of ENVI $Input $Output -b $i
done
